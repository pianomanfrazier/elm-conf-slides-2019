Random Precendence

update : Msg -> Model -> (Model, Cmd Msg)
update msg model =
    case msg of
        GetRandomClef ->
            ( model
            , Random.generate NewClef clefPool
            )

        NewClef clef ->
            ( { model | clef = clef }
            , Random.generate NewNote (clefNoteRange clef)
            )

        NewNote note ->
            ( { model | note = note }, Cmd.none )
